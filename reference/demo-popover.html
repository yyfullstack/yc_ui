Popover 弹出框测试

1. 基本用法

悬停我
// 基本用法示例代码
var basicPopover = new Popover({
    target: '#basic-popover-target',
    content: '这是一个基本的弹出框',
    trigger: 'hover'
});

2. 点击触发

点击我

4. 自定义内容

自定义内容
// 自定义内容示例代码
var customPopover = new Popover({
    target: '#custom-popover-target',
    content: function() {
        var content = document.createElement('div');
        content.innerHTML = '
自定义内容
这是一个包含自定义内容的弹出框
';
        return content;
    },
    width: '200px'
});
