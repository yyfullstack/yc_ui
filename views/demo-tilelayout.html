TileLayout 瓦片布局测试

1. 基本用法

// 基本用法示例代码
var basicTileLayout = new TileLayout({
    container: '#basic-tilelayout',
    tiles: [
        {
            title: '瓦片1',
            content: '瓦片1的内容',
            size: '1x1'
        },
        {
            title: '瓦片2',
            content: '瓦片2的内容',
            size: '2x1'
        },
        {
            title: '瓦片3',
            content: '瓦片3的内容',
            size: '1x2'
        },
        {
            title: '瓦片4',
            content: '瓦片4的内容',
            size: '2x2'
        }
    ]
});

2. 自定义瓦片

// 自定义瓦片示例代码
var customTileLayout = new TileLayout({
    container: '#custom-tilelayout',
    tiles: [
        {
            title: '自定义瓦片1',
            content: function() {
                var content = document.createElement('div');
                content.innerHTML = '
这是一个自定义内容的瓦片
';
                return content;
            },
            size: '1x1',
            backgroundColor: '#e3f2fd'
        },
        {
            title: '自定义瓦片2',
            content: '瓦片2的内容',
            size: '2x1',
            backgroundColor: '#e8f5e8'
        },
        {
            title: '自定义瓦片3',
            content: '瓦片3的内容',
            size: '1x2',
            backgroundColor: '#fff3e0'
        },
        {
            title: '自定义瓦片4',
            content: '瓦片4的内容',
            size: '2x2',
            backgroundColor: '#f3e5f5'
        }
    ]
});
